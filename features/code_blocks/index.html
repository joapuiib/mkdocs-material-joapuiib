    LongClass.java
    class LongClass {
    private int a;
    public LongClass() {
        a = 0;
    }

    public void increment() {
        a++;
    }

    public void decrement() {
        a--;
    }

    public int getA() {
        return a;
    }

    public void setA(int a) {
        this.a = a;
    }
}
